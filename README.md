# ---------- –ó–∞–≥—Ä—É–∑–∫–∞ .env –±–µ–∑ —Å—Ç–æ—Ä–æ–Ω–Ω–∏—Ö –±–∏–±–ª–∏–æ—Ç–µ–∫ ----------
def load_env(path: str = ".env") -> None:
    """
    –ü—Ä–æ—Å—Ç–µ–π—à–∞—è –∑–∞–≥—Ä—É–∑–∫–∞ –ø–µ—Ä–µ–º–µ–Ω–Ω—ã—Ö –æ–∫—Ä—É–∂–µ–Ω–∏—è –∏–∑ .env.
    –§–æ—Ä–º–∞—Ç —Å—Ç—Ä–æ–∫: KEY=VALUE, –ø—É—Å—Ç—ã–µ —Å—Ç—Ä–æ–∫–∏/–∫–æ–º–º–µ–Ω—Ç–∞—Ä–∏–∏ (#) –∏–≥–Ω–æ—Ä–∏—Ä—É—é—Ç—Å—è.
    """
    if not os.path.exists(path):
        return
    with open(path, "r", encoding="utf-8") as f:
        for line in f:
            s = line.strip()
            if not s or s.startswith("#") or "=" not in s:
                continue
            k, v = s.split("=", 1)
            os.environ.setdefault(k.strip(), v.strip())

def back_menu() -> InlineKeyboardMarkup:
    return InlineKeyboardMarkup([[InlineKeyboardButton("‚Ü©Ô∏è –ö –≥–ª–∞–≤–Ω–æ–º—É –º–µ–Ω—é", callback_data="back")]])

def products_menu() -> InlineKeyboardMarkup:
    rows: List[List[InlineKeyboardButton]] = []
    row: List[InlineKeyboardButton] = []
    for i, (name, _) in enumerate(PRODUCTS.items(), start=1):
        row.append(InlineKeyboardButton(name, callback_data=f"prod:{name}"))
        if i % 2 == 0:
            rows.append(row); row = []
    if row:
        rows.append(row)
    rows.append([InlineKeyboardButton("‚Ü©Ô∏è –ö –≥–ª–∞–≤–Ω–æ–º—É –º–µ–Ω—é", callback_data="back")])
    return InlineKeyboardMarkup(rows)

def faq_menu() -> InlineKeyboardMarkup:
    rows: List[List[InlineKeyboardButton]] = []
    for q in FAQ_ITEMS.keys():
        rows.append([InlineKeyboardButton(q, callback_data=f"faq:{q}")])
    rows.append([InlineKeyboardButton("‚Ü©Ô∏è –ö –≥–ª–∞–≤–Ω–æ–º—É –º–µ–Ω—é", callback_data="back")])
    return InlineKeyboardMarkup(rows)

# ---------- –°–æ—Å—Ç–æ—è–Ω–∏–µ –¥–ª—è –ø—Ä–∏—ë–º–∞ –æ—Ç–∑—ã–≤–æ–≤ ----------
@dataclass
class FeedbackState:
    awaiting_text: bool = False

# –ü—Ä–æ—Å—Ç–æ–µ —Ö—Ä–∞–Ω–∏–ª–∏—â–µ —Å–æ—Å—Ç–æ—è–Ω–∏–π –ø–æ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è–º
USER_STATE: Dict[int, FeedbackState] = {}

# ---------- –•—ç–Ω–¥–ª–µ—Ä—ã –∫–æ–º–∞–Ω–¥ –∏ –Ω–∞–∂–∞—Ç–∏–π ----------
async def start(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """
    /start ‚Äî –ø–µ—Ä–≤—ã–π –≤—Ö–æ–¥: –ø–æ–∫–∞–∑—ã–≤–∞–µ–º –≥–ª–∞–≤–Ω–æ–µ –º–µ–Ω—é.
    """
    await update.message.reply_text(
        "–ü—Ä–∏–≤–µ—Ç! –Ø –±–æ—Ç –ü–æ–ª—è—Ä–Ω–æ–≥–æ –ú–µ–¥–≤–µ–¥—è. –í—ã–±–µ—Ä–∏—Ç–µ –¥–µ–π—Å—Ç–≤–∏–µ:",
        reply_markup=main_menu_kbd()
    )

async def on_button(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """
    –û–±—Ä–∞–±–æ—Ç–∫–∞ –Ω–∞–∂–∞—Ç–∏–π –Ω–∞ –∏–Ω–ª–∞–π–Ω-–∫–Ω–æ–ø–∫–∏.
    """
    query = update.callback_query
    await query.answer()
    data = query.data

    if data == "back":
        await query.edit_message_text("–ì–ª–∞–≤–Ω–æ–µ –º–µ–Ω—é:", reply_markup=main_menu_kbd())
        return

    if data == "doc_links":
        text = (
            "–ü—Ä–∏–º–µ—Ä –¥–æ–∫—É–º–µ–Ω—Ç–∞—Ü–∏–∏ –∫ –ø—Ä–æ–¥—É–∫—Ç–∞–º:\n"
            f"{DOC_LINK_EXAMPLE}\n"
            f"–ü–∞—Ä–æ–ª—å: {DOC_LINK_PASSWORD}"
        )
        await query.edit_message_text(text, reply_markup=back_menu())
        return

    if data == "contacts":
        await query.edit_message_text(CONTACTS_TEXT, reply_markup=back_menu())
        return

    if data == "faq":
        await query.edit_message_text("–ö–∞–∫–æ–π –≤–æ–ø—Ä–æ—Å –≤–∞—Å –∏–Ω—Ç–µ—Ä–µ—Å—É–µ—Ç?", reply_markup=faq_menu())
        return

    if data.startswith("faq:"):
        q = data.split("faq:", 1)[1]
        ans = FAQ_ITEMS.get(q, "–û—Ç–≤–µ—Ç –Ω–µ –Ω–∞–π–¥–µ–Ω.")
        await query.edit_message_text(f"‚ùì {q}\n\nüí° {ans}", reply_markup=faq_menu())
        return

    if data == "order":
        # –ü–æ–¥—Å–∫–∞–∑–∫–∏ –ø–æ —Å—Ç–∞—Ç—É—Å—É –∑–∞–∫–∞–∑–∞ + –∫–Ω–æ–ø–∫–∞ –¥–ª—è –ø–µ—Ä–µ—Ö–æ–¥–∞ –∫ –∫–æ–Ω—Ç–∞–∫—Ç–∞–º
        kbd = InlineKeyboardMarkup([
            [InlineKeyboardButton("–ö–æ–Ω—Ç–∞–∫—Ç–Ω–∞—è –∏–Ω—Ñ–æ—Ä–º–∞—Ü–∏—è", callback_data="contacts")],
            [InlineKeyboardButton("‚Ü©Ô∏è –ö –≥–ª–∞–≤–Ω–æ–º—É –º–µ–Ω—é", callback_data="back")],
        ])
        await query.edit_message_text(
            "–ß—Ç–æ–±—ã –ø—Ä–æ–≤–µ—Ä–∏—Ç—å —Å—Ç–∞—Ç—É—Å –∑–∞–∫–∞–∑–∞, –æ—Ç–ø—Ä–∞–≤—å—Ç–µ –º–µ–Ω–µ–¥–∂–µ—Ä—É:\n"
            "‚Ä¢ –ù–∞–∏–º–µ–Ω–æ–≤–∞–Ω–∏–µ —Ç–æ–≤–∞—Ä–∞\n‚Ä¢ –í–∞—à–µ –∏–º—è\n‚Ä¢ –ö–æ–Ω—Ç–∞–∫—Ç—ã –¥–ª—è —Å–≤—è–∑–∏\n\n"
            "–ö–æ–Ω—Ç–∞–∫—Ç—ã ‚Äî –ø–æ –∫–Ω–æ–ø–∫–µ –Ω–∏–∂–µ.",
            reply_markup=kbd
        )
        return

    if data == "feedback":
        # –ü—Ä–æ—Å–∏–º –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è –æ—Ç–ø—Ä–∞–≤–∏—Ç—å –æ—Ç–∑—ã–≤ –æ–¥–Ω–∏–º —Å–æ–æ–±—â–µ–Ω–∏–µ–º
        uid = query.from_user.id
        USER_STATE[uid] = FeedbackState(awaiting_text=True)
        await query.edit_message_text(
            "–ù–∞–ø–∏—à–∏—Ç–µ –≤–∞—à –æ—Ç–∑—ã–≤ –æ–¥–Ω–∏–º —Å–æ–æ–±—â–µ–Ω–∏–µ–º. –ü–æ—Å–ª–µ –æ—Ç–ø—Ä–∞–≤–∫–∏ –ø–æ–ª—É—á–∏—Ç–µ –ø–æ–¥—Ç–≤–µ—Ä–∂–¥–µ–Ω–∏–µ.",
            reply_markup=back_menu()
        )
        return

    if data == "products":
        await query.edit_message_text("–°–ø–∏—Å–æ–∫ –ø—Ä–æ–¥—É–∫—Ç–æ–≤:", reply_markup=products_menu())
        return

    if data.startswith("prod:"):
        name = data.split("prod:", 1)[1]
        link = PRODUCTS.get(name, "–°—Å—ã–ª–∫–∞ –Ω–µ –Ω–∞–π–¥–µ–Ω–∞.")
        await query.edit_message_text(f"{name} ‚Äî –¥–æ–∫—É–º–µ–Ω—Ç–∞—Ü–∏—è:\n{link}", reply_markup=products_menu())
        return

    if data == "assist":
        await query.edit_message_text(
            "–î–ª—è –ø–æ–¥–±–æ—Ä–∞ —Ä–µ—à–µ–Ω–∏—è –Ω–∞–ø–∏—à–∏—Ç–µ –≤ —Ç–µ—Ö–ø–æ–¥–¥–µ—Ä–∂–∫—É: @PBPOLAR",
            reply_markup=back_menu()
        )
        return

    if data == "forum":
        await query.edit_message_text(f"–§–æ—Ä—É–º —Ä–∞–∑—Ä–∞–±–æ—Ç—á–∏–∫–æ–≤: {FORUM_LINK}", reply_markup=back_menu())
        return

    if data == "news":
        await query.edit_message_text(NEWS_TEXT, reply_markup=back_menu())
        return

async def on_message(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """
    –ü—Ä–∏—ë–º —Å–≤–æ–±–æ–¥–Ω—ã—Ö —Å–æ–æ–±—â–µ–Ω–∏–π. –ï—Å–ª–∏ –≤–∫–ª—é—á—ë–Ω —Ä–µ–∂–∏–º ¬´–æ–∂–∏–¥–∞–Ω–∏—è –æ—Ç–∑—ã–≤–∞¬ª, –ø–µ—Ä–µ—Å—ã–ª–∞–µ–º –∞–¥–º–∏–Ω—É.
    –ò–Ω–∞—á–µ ‚Äî –ø–æ–∫–∞–∑—ã–≤–∞–µ–º –≥–ª–∞–≤–Ω–æ–µ –º–µ–Ω—é.
    """
    uid = update.effective_user.id
    st = USER_STATE.get(uid)

    # –ï—Å–ª–∏ –∂–¥—ë–º –æ—Ç–∑—ã–≤ ‚Äî –ø–µ—Ä–µ—Å—ã–ª–∞–µ–º –∞–¥–º–∏–Ω—É –∏ —Å–±—Ä–∞—Å—ã–≤–∞–µ–º —Å–æ—Å—Ç–æ—è–Ω–∏–µ
    if st and st.awaiting_text and update.message and update.message.text:
        text = update.message.text.strip()
        if ADMIN_USER_ID:
            try:
                await context.bot.send_message(
                    chat_id=ADMIN_USER_ID,
                    text=f"üì© –ù–æ–≤—ã–π –æ—Ç–∑—ã–≤\nüë§ –ü–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—å: {uid}\nüí¨ {text}",
                )
            except Exception:
                # –û—à–∏–±–∫–∏ –¥–æ—Å—Ç–∞–≤–∫–∏ –∞–¥–º–∏–Ω—É –∏–≥–Ω–æ—Ä–∏—Ä—É–µ–º, —á—Ç–æ–±—ã –Ω–µ —Ç—Ä–µ–≤–æ–∂–∏—Ç—å –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è
                pass
        USER_STATE[uid] = FeedbackState(awaiting_text=False)
        await update.message.reply_text("–°–ø–∞—Å–∏–±–æ! –í–∞—à –æ—Ç–∑—ã–≤ –æ—Ç–ø—Ä–∞–≤–ª–µ–Ω. üôå")
        await update.message.reply_text("–ì–ª–∞–≤–Ω–æ–µ –º–µ–Ω—é:", reply_markup=main_menu_kbd())
        return

    # –õ—é–±–æ–π –¥—Ä—É–≥–æ–π —Ç–µ–∫—Å—Ç ‚Äî –ø—Ä–æ—Å—Ç–æ –ø–æ–∫–∞–∑–∞—Ç—å –≥–ª–∞–≤–Ω–æ–µ –º–µ–Ω—é
    await update.message.reply_text("–ì–ª–∞–≤–Ω–æ–µ –º–µ–Ω—é:", reply_markup=main_menu_kbd())

def main():
    """
    –¢–æ—á–∫–∞ –≤—Ö–æ–¥–∞: —Å–æ–±–∏—Ä–∞–µ–º Application, —Ä–µ–≥–∏—Å—Ç—Ä–∏—Ä—É–µ–º —Ö—ç–Ω–¥–ª–µ—Ä—ã –∏ –∑–∞–ø—É—Å–∫–∞–µ–º polling.
    """
    if not BOT_TOKEN:
        raise SystemExit("–ù–µ –∑–∞–¥–∞–Ω BOT_TOKEN (—Å–º. —Ñ–∞–π–ª .env)")
    app = ApplicationBuilder().token(BOT_TOKEN).build()
    app.add_handler(CommandHandler("start", start))
    app.add_handler(CallbackQueryHandler(on_button))
    app.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND, on_message))
    app.run_polling(close_loop=False)

if __name__ == "__main__":
    asyncio.run(main())
